#include "include/push.h"

int	ft_count_args(char **str)
{
	int	j;

	j = 0;
	while (str[j])
		j++;
	return (j);
}

void	ft_start_1(char **av, t_list *a, t_list	*b)
{
	int		j;
	int		ac;
	t_list	*new;
	char	**str;
	
	new = NULL;
	j = 1;
	str = ft_strcheck(av);
	ac = ft_count_args(str);
	while (j < ac)
	{
		ft_checkisnumber(str[j]);
		j++;
	}
	ft_checkdouble(ac, str);
	ft_buildstacka(ac, str, &a, &new);
	ft_sorting(ac, &a, &b);
	ft_printlist(a);
	free(str);
	return ;
}
void	ft_start_2(int ac, char **av, t_list *a, t_list	*b)
{
	int		j;
	t_list	*new;

	new = NULL;
	j = 1;
	while (j < ac)
	{
		ft_checkisnumber(av[j]);
		j++;
	}
	ft_checkdouble(ac, av);
	ft_buildstacka(ac, av, &a, &new);
	ft_sorting(ac, &a, &b);
	ft_printlist(a);
	return ;
}

int	main(int ac, char **av)
{
	t_list	*b;
	t_list	*a;
	
	a = NULL;	
	b = NULL;
	if (ac == 1)
		return (0);
	if (ac == 2)
		ft_start_1(av, a, b);
	else
	ft_start_2(ac, av, a, b);	
	ft_free_stack(a);
	ft_free_stack(b);
}
